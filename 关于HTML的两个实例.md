代码如下:

    <!DOCTYPE html>
    <html>
    <head lang="en">
        <meta charset="UTF-8">
        <title>菜谱</title>
    </head>
    <body>
    <table border="1">
        <tr>
            <th colspan="5">星期一菜谱</th>
        </tr>
        <tr>
            <th rowspan="2">素菜</th>
            <th colspan="3">清炒茄子</th>
            <th>花椒扁豆</th>
        </tr>
        <tr>
            <th colspan="3">小葱豆腐</th>
            <th>炒白菜</th>
        </tr>
        <tr>
            <th rowspan="2">荤菜</th>
            <th colspan="3">油闷大虾</th>
            <th>海参鱼翅</th>
        </tr>
        <tr>

            <th>烤全羊</th>
        </tr>
    </table>
    </body>
    </html>
